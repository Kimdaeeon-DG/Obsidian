int ft_atoi(const char *str)

{

int sign;

long long result;

  

sign = 1;

result = 0;

while ((*str >= 9 && *str <= 13) || *str == 32)

str++;

if (*str == '-')

sign *= -1;

if (*str == '-' || *str == '+')

str++;

while (*str >= '0' && *str <= '9')

{

result = result * 10 + (*str - '0');

str++;

}

result = result * sign;

return ((int)result);

}
